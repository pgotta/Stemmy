#   pip install -r requirements.txt
#   pip install --force-reinstall --no-cache-dir torch torchaudio --index-url https://download.pytorch.org/whl/cu128
#
# Verify the GPU is seen:  python -c "import torch; print(torch.cuda.is_available())"
# (Blackwell sm_120 needs cu128 or newer; on a different GPU swap the index URL.)
